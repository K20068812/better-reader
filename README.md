SETUP BACKEND:
1. cd better-reader-master
2. pip install -r requirements.txt
3. python manage.py makemigrations
4. python manage.py migrate
5. python manage.py runserver

SETUP FRONTEND:
1. cd client
2. npm i
3. npm run dev
